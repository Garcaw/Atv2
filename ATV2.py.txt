class Lista:
    def __init__(self):
        self.dados = []

    def inserir(self, numero):
        self.dados.append(numero)

    def remover_primeiro(self):
        if self.dados:
            return self.dados.pop(0)
        else:
            return None

class Pilha:
    def __init__(self):
        self.dados = []

    def inserir(self, numero):
        self.dados.append(numero)

    def remover(self):
        if self.dados:
            return self.dados.pop()
        else:
            return None

class Fila:
    def __init__(self):
        self.dados = []

    def inserir(self, numero):
        self.dados.append(numero)

    def remover(self):
        if self.dados:
            return self.dados.pop(0)
        else:
            return None

# Passo 1
lista = Lista()
for i in range(1, 6):
